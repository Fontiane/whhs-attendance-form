  <script>

    function submitForm() {

      const date = document.getElementById('attDate').value;

      const players = Array.from(document.querySelectorAll('input[type="checkbox"]:checked'))

        .map(cb => cb.value);


      if (!date) {

        alert("Please select a date.");

        return;

      }


      fetch('https://your_power_automate_webhook_url_here', {

        method: 'POST',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify({ date, players })

      })

      .then(res => res.ok ? alert("Attendance submitted!") : alert("Submission failed"))

      .catch(() => alert("Error sending attendance"));

    }
